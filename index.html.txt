// Obtener elementos
const modal = document.getElementById('modal');
const modalTitle = document.getElementById('modal-title');
const modalContent = document.getElementById('modal-content');
const closeBtn = document.getElementById('closeModal');

// Añadir evento a cada botón
document.querySelectorAll('nav button').forEach(button => {
button.addEventListener('click', () => {
const sectionKey = button.getAttribute('data-section');
showModal(sectionKey);
});
});

// Función para mostrar modal
function showModal(sectionKey) {
const section = sections[sectionKey];
if (section) {
modalTitle.innerText = section.title;
modalContent.innerHTML = section.content;
modal.style.display = 'block';
}
}

// Función para cerrar modal
closeBtn.addEventListener('click', () => {
modal.style.display = 'none';
});

// Cerrar si se hace clic fuera de la ventana modal
window.addEventListener('click', (event) => {
if (event.target == modal) {
modal.style.display = 'none';
}
});

// Funciones para gestionar el banner de cookies
const cookieBanner = document.getElementById('cookie-banner');
const acceptCookiesBtn = document.getElementById('accept-cookies');

// Función para comprobar si el usuario ha aceptado cookies
function hasAcceptedCookies() {
return document.cookie.split(';').some(cookie => cookie.trim().startsWith('cookiesAccepted='));
}

// Mostrar banner si no ha aceptado
if (!hasAcceptedCookies()) {
cookieBanner.style.display = 'flex';
}

// Evento para aceptar cookies
acceptCookiesBtn.addEventListener('click', () => {
// Guardar en cookies
document.cookie = "cookiesAccepted=true; path=/; max-age=" + (60*60*24*365);
// Ocultar banner
cookieBanner.style.display = 'none';
});
